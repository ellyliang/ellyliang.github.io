文件目录结构

. ├── index.html └── static └── js ├── bookStoreMain.js └── lib ├── angular-route.js ├── angular.js ├── app.js ├── config.js ├── domReady.js ├── jquery.js └── require.js

index.html

``` 简单requirejs&ng的demo

{{name || '柚子－use...'}} 

    <script data-main="/static/js/lib/config.js" src="/static/js/lib/require.js"></script>
    <script src="/static/js/bookStoreMain.js"></script>
</body>
</html>

```

Note: data-main是requirejs引入模块的路径

app.js

define([ 'angular', 'ngRoute' ], function(angular, ngRoute) { var app = angular.module('bookStore', ['ngRoute']); return app; });

config.js

require.config({ paths : { "angular" : "angular", "domReady" : "domReady", "ngRoute" : "angular-route", "jquery" : "jquery", "app" : "app" }, shim : { 'angular' : { 'exports' : 'angular' }, 'ngRoute' : ['angular'], 'app' : { 'exports' : 'app' } } });

bookStoreMain.js

require([ 'angular', 'domReady!', 'jquery', 'app' ], function(angular, document, jquery, app) { angular.bootstrap(document.body, ["bookStore"]); });
